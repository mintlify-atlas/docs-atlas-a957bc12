Motion is a simple yet powerful animation library for JavaScript and React. This guide will help you create your first animation.

## Installation

<Steps>
  <Step title="Install Motion">
    <Tabs>
      <Tab title="React">
        ```bash
        npm install motion
        ```
      </Tab>
      <Tab title="JavaScript">
        ```bash
        npm install motion
        ```
      </Tab>
    </Tabs>
  </Step>

  <Step title="Import Motion">
    <Tabs>
      <Tab title="React">
        ```jsx
        import { motion } from "motion/react"
        ```
      </Tab>
      <Tab title="JavaScript">
        ```javascript
        import { animate } from "motion"
        ```
      </Tab>
    </Tabs>
  </Step>

  <Step title="Create your first animation">
    <Tabs>
      <Tab title="React">
        Add the `animate` prop to a `motion` component to animate it on mount:

        ```jsx
        function App() {
          return (
            <motion.div
              animate={{ x: 100 }}
              style={{ width: 100, height: 100, background: "#0077ff" }}
            />
          )
        }
        ```

        The `motion.div` will slide 100 pixels to the right when it appears.
      </Tab>
      <Tab title="JavaScript">
        Use the `animate` function to animate any element:

        ```javascript
        import { animate } from "motion"

        animate("#box", { x: 100 })
        ```

        ```html
        <div id="box" style="width: 100px; height: 100px; background: #0077ff;"></div>
        ```

        The element will slide 100 pixels to the right.
      </Tab>
    </Tabs>
  </Step>
</Steps>

## Add interactivity

Make your animations interactive with hover effects and springs.

<Tabs>
  <Tab title="React">
    Use `whileHover` to animate on hover, and add a spring transition for natural motion:

    ```jsx
    function App() {
      return (
        <motion.div
          whileHover={{ scale: 1.2 }}
          transition={{ type: "spring", bounce: 0.4 }}
          style={{ 
            width: 100, 
            height: 100, 
            background: "#0077ff",
            borderRadius: 10
          }}
        />
      )
    }
    ```

    The box will spring and scale up when you hover over it.
  </Tab>
  <Tab title="JavaScript">
    Listen to events and trigger animations with springs:

    ```javascript
    import { animate } from "motion"

    const box = document.querySelector("#box")

    box.addEventListener("mouseenter", () => {
      animate(box, 
        { scale: 1.2 }, 
        { type: "spring", bounce: 0.4 }
      )
    })

    box.addEventListener("mouseleave", () => {
      animate(box, 
        { scale: 1 }, 
        { type: "spring", bounce: 0.4 }
      )
    })
    ```

    The box will spring and scale when you hover over it.
  </Tab>
</Tabs>

## Control animations with state

<Tabs>
  <Tab title="React">
    Animate based on component state:

    ```jsx
    import { motion } from "motion/react"
    import { useState } from "react"

    function App() {
      const [isActive, setIsActive] = useState(false)

      return (
        <motion.div
          animate={{ x: isActive ? 200 : 0 }}
          transition={{ duration: 0.4 }}
          onClick={() => setIsActive(!isActive)}
          style={{ 
            width: 100, 
            height: 100, 
            background: "#0077ff",
            cursor: "pointer"
          }}
        />
      )
    }
    ```

    Click the box to toggle its position.
  </Tab>
  <Tab title="JavaScript">
    Control animations programmatically:

    ```javascript
    import { animate } from "motion"

    const box = document.querySelector("#box")
    let isActive = false

    box.addEventListener("click", () => {
      isActive = !isActive
      animate(box, 
        { x: isActive ? 200 : 0 }, 
        { duration: 0.4 }
      )
    })
    ```

    Click the box to toggle its position.
  </Tab>
</Tabs>
